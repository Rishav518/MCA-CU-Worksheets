#!/bin/bash

# Check if input directory exists
if [ ! -d "$1" ]; then
  echo "Error: input directory does not exist."
  exit 1
fi

# Create folders a, b and c within input directory
mkdir "$1/a" "$1/b" "$1/c"

# Task 1: Store hostname in a text file in folder a
hostname=$(hostname)
echo $hostname > "$1/a/hostname.txt"

# Task 2: Store structure of files in C drive in folder b
ls -R /c/ > "$1/b/c_files_structure.txt"

# Task 3: Move folders a and b with their content to folder c
mv "$1/a" "$1/b" "$1/c"
